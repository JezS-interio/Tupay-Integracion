rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Users collection - allow users to read/write their own profile
    match /users/{userId} {
      allow read: if request.auth != null;
      allow create: if request.auth != null;
      allow update: if request.auth != null && request.auth.uid == userId;
      allow delete: if request.auth != null && request.auth.uid == userId;
    }

    // User carts collection - allow users to read/write their own cart
    match /user_carts/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }

    // Abandoned carts collection - allow users to read/write their own abandoned carts
    match /abandoned_carts/{email} {
      allow read, write: if request.auth != null && request.auth.token.email == email;
    }

    // Products collection - read for all, write for authenticated users (for now)
    match /products/{productId} {
      allow read: if true;
      allow write: if request.auth != null;
    }

    // Orders collection (for future use)
    match /orders/{orderId} {
      allow read: if request.auth != null && resource.data.userId == request.auth.uid;
      allow create: if request.auth != null;
      allow update: if request.auth != null && resource.data.userId == request.auth.uid;
    }

    // Newsletter subscribers - allow anyone to subscribe (server-side write)
    match /newsletter_subscribers/{email} {
      allow read, write: if true;
    }
  }
}
